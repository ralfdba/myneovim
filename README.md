# My neovim
My custom neovim config (for Mac M1)
# Steps
* brew install neovim
* mkdir ~/.config/nvim
* ln -s $(which nvim) /opt/homebrew/bin/vim
# Restart ITerm
touch ~/.config/nvim/init.vim
